# 任务依赖关系支持设计方案

## 一、概述

为现有的多线程框架设计方案增加全面的任务依赖关系支持功能，允许用户定义不同任务之间的执行顺序和依赖关系，确保在多线程环境下任务能够按照预设的依赖关系正确执行。

## 二、核心组件设计

#### 2.1.3 DependentTask 支持依赖关系的任务类

```csharp
namespace Basement.Threading
{
    public class DependentTask : ITaskDependency
    {
        private readonly ITask _task;
        private readonly List<ITaskDependency> _dependencies = new List<ITaskDependency>();
        private readonly List<ITaskDependency> _dependents = new List<ITaskDependency>();
        private TaskStatus _status = TaskStatus.Created;
        private Exception _exception = null;
        private readonly object _lock = new object();

        public string TaskId { get; }
        public TaskStatus Status
        {
            get { return _status; }
            internal set
            {
                lock (_lock)
                {
                    if (_status != value)
                    {
                        TaskStatus oldStatus = _status;
                        _status = value;
                        OnStatusChanged?.Invoke(this, oldStatus, _status);
                    }
                }
            }
        }

        public IReadOnlyCollection<ITaskDependency> Dependencies
        {
            get { return _dependencies.AsReadOnly(); }
        }

        public IReadOnlyCollection<ITaskDependency> Dependents
        {
            get { return _dependents.AsReadOnly(); }
        }

        public Exception Exception
        {
            get { return _exception; }
            internal set { _exception = value; }
        }

        public TaskPriority Priority
        {
            get { return _task.Priority; }
        }

        public Action OnComplete
        {
            get { return _task.OnComplete; }
            set { _task.OnComplete = value; }
        }

        public Action<ITaskDependency, TaskStatus, TaskStatus> OnStatusChanged { get; set; }

        public DependentTask(ITask task, string taskId = null)
        {
            _task = task;
            TaskId = taskId ?? Guid.NewGuid().ToString();
        }

        public void AddDependency(ITaskDependency dependency)
        {
            lock (_lock)
            {
                if (dependency == null || dependency == this)
                    throw new ArgumentException("Invalid dependency!");

                if (!_dependencies.Contains(dependency))
                {
                    _dependencies.Add(dependency);
                    dependency._dependents.Add(this);
                    UpdateStatus();
                }
            }
        }

        public bool RemoveDependency(ITaskDependency dependency)
        {
            lock (_lock)
            {
                if (_dependencies.Remove(dependency))
                {
                    dependency._dependents.Remove(this);
                    UpdateStatus();
                    return true;
                }
                return false;
            }
        }

        public void Execute()
        {
            try
            {
                Status = TaskStatus.Running;
                _task.Execute();
                Status = TaskStatus.Completed;
            }
            catch (Exception ex)
            {
                Exception = ex;
                Status = TaskStatus.Failed;
                throw;
            }
            finally
            {
                OnComplete?.Invoke();
            }
        }

        /// <summary>
        /// 更新任务状态
        /// </summary>
        internal void UpdateStatus()
        {
            lock (_lock)
            {
                if (_status == TaskStatus.Running || _status == TaskStatus.Completed || _status == TaskStatus.Failed || _status == TaskStatus.Canceled)
                    return;

                // 检查所有依赖任务是否已完成
                bool allDependenciesCompleted = _dependencies.All(dep => dep.Status == TaskStatus.Completed);
                
                if (allDependenciesCompleted)
                {
                    Status = TaskStatus.Ready;
                }
                else
                {
                    Status = TaskStatus.Waiting;
                }
            }
        }
    }
}
```

#### 2.1.3 TaskDependencyManager 任务依赖管理器

```csharp
namespace Basement.Threading
{
    public class TaskDependencyManager
    {
        private readonly Dictionary<string, ITaskDependency> _tasks = new Dictionary<string, ITaskDependency>();
        private readonly object _lock = new object();
        private readonly TaskScheduler _taskScheduler;

        public TaskDependencyManager(TaskScheduler taskScheduler)
        {
            _taskScheduler = taskScheduler;
        }

        /// <summary>
        /// 添加任务
        /// </summary>
        public void AddTask(ITaskDependency task)
        {
            lock (_lock)
            {
                if (!_tasks.ContainsKey(task.TaskId))
                {
                    _tasks[task.TaskId] = task;
                    task.OnStatusChanged += OnTaskStatusChanged;
                    task.UpdateStatus();
                }
            }
        }

        /// <summary>
        /// 获取任务
        /// </summary>
        public ITaskDependency GetTask(string taskId)
        {
            lock (_lock)
            {
                _tasks.TryGetValue(taskId, out ITaskDependency task);
                return task;
            }
        }

        /// <summary>
        /// 移除任务
        /// </summary>
        public bool RemoveTask(string taskId)
        {
            lock (_lock)
            {
                if (_tasks.TryGetValue(taskId, out ITaskDependency task))
                {
                    // 移除所有依赖关系
                    foreach (var dependency in task.Dependencies.ToList())
                    {
                        task.RemoveDependency(dependency);
                    }
                    
                    foreach (var dependent in task.Dependents.ToList())
                    {
                        dependent.RemoveDependency(task);
                    }
                    
                    task.OnStatusChanged -= OnTaskStatusChanged;
                    _tasks.Remove(taskId);
                    return true;
                }
                return false;
            }
        }

        /// <summary>
        /// 创建任务依赖关系
        /// </summary>
        public void CreateDependency(string dependentTaskId, string dependencyTaskId)
        {
            lock (_lock)
            {
                if (_tasks.TryGetValue(dependentTaskId, out ITaskDependency dependentTask) &&
                    _tasks.TryGetValue(dependencyTaskId, out ITaskDependency dependencyTask))
                {
                    dependentTask.AddDependency(dependencyTask);
                }
            }
        }

        /// <summary>
        /// 移除任务依赖关系
        /// </summary>
        public bool RemoveDependency(string dependentTaskId, string dependencyTaskId)
        {
            lock (_lock)
            {
                if (_tasks.TryGetValue(dependentTaskId, out ITaskDependency dependentTask) &&
                    _tasks.TryGetValue(dependencyTaskId, out ITaskDependency dependencyTask))
                {
                    return dependentTask.RemoveDependency(dependencyTask);
                }
                return false;
            }
        }

        /// <summary>
        /// 任务状态变化回调
        /// </summary>
        private void OnTaskStatusChanged(ITaskDependency task, TaskStatus oldStatus, TaskStatus newStatus)
        {
            // 更新所有依赖此任务的后继任务状态
            foreach (var dependent in task.Dependents)
            {
                dependent.UpdateStatus();
            }

            // 如果任务已就绪，提交到调度器执行
            if (newStatus == TaskStatus.Ready)
            {
                _taskScheduler.SubmitTask(task);
            }
            
            // 处理任务执行失败的情况
            if (newStatus == TaskStatus.Failed)
            {
                HandleTaskFailure(task);
            }
        }

        /// <summary>
        /// 处理任务执行失败的情况
        /// </summary>
        private void HandleTaskFailure(ITaskDependency failedTask)
        {
            // 默认行为：将所有依赖此任务的后继任务标记为失败
            foreach (var dependent in failedTask.Dependents)
            {
                dependent.Status = TaskStatus.Failed;
            }
        }

        /// <summary>
        /// 检查循环依赖
        /// </summary>
        public bool HasCircularDependency(out List<string> circularPath)
        {
            lock (_lock)
            {
                var visited = new HashSet<string>();
                var recursionStack = new HashSet<string>();
                var path = new List<string>();
                circularPath = null;

                foreach (var taskId in _tasks.Keys)
                {
                    if (!visited.Contains(taskId))
                    {
                        if (HasCircularDependencyUtil(taskId, visited, recursionStack, path, out circularPath))
                        {
                            return true;
                        }
                    }
                }
                return false;
            }
        }

        /// <summary>
        /// 检查循环依赖的辅助方法
        /// </summary>
        private bool HasCircularDependencyUtil(string taskId, HashSet<string> visited, HashSet<string> recursionStack, List<string> path, out List<string> circularPath)
        {
            circularPath = null;

            if (!visited.Contains(taskId))
            {
                visited.Add(taskId);
                recursionStack.Add(taskId);
                path.Add(taskId);

                var task = _tasks[taskId];
                foreach (var dependency in task.Dependencies)
                {
                    if (!visited.Contains(dependency.TaskId))
                    {
                        if (HasCircularDependencyUtil(dependency.TaskId, visited, recursionStack, path, out circularPath))
                        {
                            return true;
                        }
                    }
                    else if (recursionStack.Contains(dependency.TaskId))
                    {
                        // 找到循环依赖，构建循环路径
                        int index = path.IndexOf(dependency.TaskId);
                        if (index != -1)
                        {
                            circularPath = path.GetRange(index, path.Count - index);
                            circularPath.Add(dependency.TaskId); // 闭合循环
                        }
                        return true;
                    }
                }
            }

            recursionStack.Remove(taskId);
            path.RemoveAt(path.Count - 1);
            return false;
        }

        /// <summary>
        /// 拓扑排序，获取任务执行顺序
        /// </summary>
        public List<ITaskDependency> TopologicalSort()
        {
            lock (_lock)
            {
                var result = new List<ITaskDependency>();
                var inDegree = new Dictionary<string, int>();
                var queue = new Queue<ITaskDependency>();

                // 计算每个任务的入度（依赖任务数量）
                foreach (var task in _tasks.Values)
                {
                    int degree = task.Dependencies.Count;
                    inDegree[task.TaskId] = degree;
                    if (degree == 0)
                    {
                        queue.Enqueue(task);
                    }
                }

                // 执行拓扑排序
                while (queue.Count > 0)
                {
                    var current = queue.Dequeue();
                    result.Add(current);

                    foreach (var dependent in current.Dependents)
                    {
                        inDegree[dependent.TaskId]--;
                        if (inDegree[dependent.TaskId] == 0)
                        {
                            queue.Enqueue(dependent);
                        }
                    }
                }

                // 检查是否存在循环依赖
                if (result.Count != _tasks.Count)
                {
                    throw new InvalidOperationException("Circular dependency detected!");
                }

                return result;
            }
        }

        /// <summary>
        /// 清空所有任务
        /// </summary>
        public void Clear()
        {
            lock (_lock)
            {
                foreach (var task in _tasks.Values)
                {
                    task.OnStatusChanged -= OnTaskStatusChanged;
                }
                _tasks.Clear();
            }
        }
    }
}
```

```csharp
            }

        /// </summary>

        {
            lock (_lock)
            {
                return _dependencyManager.HasCircularDependency(out circularPath);
            }
        }

        /// <summary>
        /// 拓扑排序，获取任务执行顺序
        /// </summary>
        public List<ITaskDependency> GetExecutionOrder()
        {
            lock (_lock)
            {
                return _dependencyManager.TopologicalSort();
            }
        }

        /// <summary>
        /// 提交任务图，自动处理依赖关系
        /// </summary>
        public void SubmitTaskGraph(params ITaskDependency[] tasks)
        {
            lock (_lock)
            {
                // 添加所有任务
                foreach (var task in tasks)
                {
                    _dependencyManager.AddTask(task);
                }

                // 检查循环依赖
                if (_dependencyManager.HasCircularDependency(out List<string> circularPath))
                {
                    string path = string.Join(" -> ", circularPath);
                    throw new InvalidOperationException($"Circular dependency detected: {path}");
                }

                // 提交所有就绪的任务
                foreach (var task in tasks)
                {
                    if (task.Status == TaskStatus.Ready)
                    {
                        SubmitTask(task);
                    }
                }
            }
        }
    }
}
```

### 2.3 依赖关系可视化功能

```csharp
namespace Basement.Threading
{
    public class TaskDependencyVisualizer
    {
        private readonly TaskDependencyManager _dependencyManager;
        private bool _isVisible = false;
        private Vector2 _scrollPosition;
        private readonly Dictionary<string, Vector2> _nodePositions = new Dictionary<string, Vector2>();
        private readonly float _nodeWidth = 200f;
        private readonly float _nodeHeight = 50f;
        private readonly float _nodeSpacing = 50f;

        public TaskDependencyVisualizer(TaskDependencyManager dependencyManager)
        {
            _dependencyManager = dependencyManager;
        }

        public void ToggleVisibility()
        {
            _isVisible = !_isVisible;
        }

        public void OnGUI()
        {
            if (!_isVisible)
                return;

            GUILayout.Window(100, new Rect(10, 10, Screen.width - 20, Screen.height - 20), DrawWindow, "任务依赖关系可视化");
        }

        private void DrawWindow(int windowId)
        {
            GUILayout.BeginHorizontal(EditorStyles.toolbar);
            GUILayout.Label("任务依赖关系图", EditorStyles.toolbarButton);
            GUILayout.FlexibleSpace();
            if (GUILayout.Button("关闭", EditorStyles.toolbarButton))
            {
                _isVisible = false;
            }
            GUILayout.EndHorizontal();

            _scrollPosition = GUILayout.BeginScrollView(_scrollPosition, true, true);
            
            // 绘制任务节点和依赖关系
            DrawTaskGraph();
            
            GUILayout.EndScrollView();

            GUI.DragWindow();
        }

        private void DrawTaskGraph()
        {
            // 使用拓扑排序确定节点位置
            var sortedTasks = _dependencyManager.TopologicalSort();
            
            // 计算节点位置
            CalculateNodePositions(sortedTasks);
            
            // 绘制依赖关系线
            DrawDependencyLines();
            
            // 绘制任务节点
            DrawTaskNodes();
        }

        private void CalculateNodePositions(List<ITaskDependency> sortedTasks)
        {
            // 简单的水平布局算法
            float x = _nodeSpacing;
            float y = _nodeSpacing;
            float maxY = y;

            foreach (var task in sortedTasks)
            {
                if (!_nodePositions.ContainsKey(task.TaskId))
                {
                    _nodePositions[task.TaskId] = new Vector2(x, y);
                    y += _nodeHeight + _nodeSpacing;
                    
                    if (y > maxY)
                    {
                        maxY = y;
                    }
                    
                    // 换行
                    if (y > Screen.height - _nodeHeight - _nodeSpacing * 2)
                    {
                        x += _nodeWidth + _nodeSpacing * 2;
                        y = _nodeSpacing;
                    }
                }
            }
        }

        private void DrawDependencyLines()
        {
            foreach (var task in _dependencyManager.TopologicalSort())
            {
                if (_nodePositions.TryGetValue(task.TaskId, out Vector2 taskPos))
                {
                    foreach (var dependency in task.Dependencies)
                    {
                        if (_nodePositions.TryGetValue(dependency.TaskId, out Vector2 depPos))
                        {
                            // 计算连线起点和终点
                            Vector2 start = new Vector2(depPos.x + _nodeWidth, depPos.y + _nodeHeight / 2);
                            Vector2 end = new Vector2(taskPos.x, taskPos.y + _nodeHeight / 2);
                            
                            // 绘制连线
                            Handles.color = Color.gray;
                            Handles.DrawLine(start, end);
                            
                            // 绘制箭头
                            Vector2 direction = end - start;
                            float arrowSize = 10f;
                            Vector2 arrow1 = end - direction.normalized * arrowSize + Vector3.Cross(direction.normalized, Vector3.forward) * arrowSize / 2;
                            Vector2 arrow2 = end - direction.normalized * arrowSize - Vector3.Cross(direction.normalized, Vector3.forward) * arrowSize / 2;
                            
                            Handles.DrawLine(end, arrow1);
                            Handles.DrawLine(end, arrow2);
                        }
                    }
                }
            }
        }

        private void DrawTaskNodes()
        {
            foreach (var task in _dependencyManager.TopologicalSort())
            {
                if (_nodePositions.TryGetValue(task.TaskId, out Vector2 pos))
                {
                    // 根据任务状态设置颜色
                    Color color = GetTaskStatusColor(task.Status);
                    
                    // 绘制节点背景
                    GUI.color = color;
                    GUI.Box(new Rect(pos.x, pos.y, _nodeWidth, _nodeHeight), GUIContent.none);
                    
                    // 绘制节点内容
                    GUI.color = Color.white;
                    GUILayout.BeginArea(new Rect(pos.x + 5, pos.y + 5, _nodeWidth - 10, _nodeHeight - 10));
                    GUILayout.Label($"Task: {task.TaskId.Substring(0, 8)}...", EditorStyles.boldLabel);
                    GUILayout.Label($"Status: {task.Status}");
                    GUILayout.Label($"Priority: {task.Priority}");
                    GUILayout.EndArea();
                }
            }
        }

        private Color GetTaskStatusColor(TaskStatus status)
        {
            switch (status)
            {
                case TaskStatus.Created:
                    return Color.gray;
                case TaskStatus.Waiting:
                    return Color.yellow;
                case TaskStatus.Ready:
                    return Color.green;
                case TaskStatus.Running:
                    return Color.blue;
                case TaskStatus.Completed:
                    return Color.green;
                case TaskStatus.Failed:
                    return Color.red;
                case TaskStatus.Canceled:
                    return Color.orange;
                default:
                    return Color.gray;
            }
        }
    }
}
```

```csharp
namespace Basement.Threading

    public class TaskDependencyManager

            if (_failureStrategy != null)
```

```csharp
    public void RunLargeTaskGraphTest(int taskCount, int averageDependenciesPerTask)
    {
        var stopwatch = Stopwatch.StartNew();
        
        // 创建大量任务
        var tasks = new List<ITaskDependency>();
        for (int i = 0; i < taskCount; i++)
        {
            var task = TaskScheduler.Instance.CreateDependentTask(() => {
                // 模拟任务执行
                Thread.Sleep(10);
            }, TaskPriority.Normal, $"Task_{i}");
            tasks.Add(task);
        }
        
        // 随机创建依赖关系
        var random = new Random();
        for (int i = 0; i < taskCount; i++)
        {
            int dependencyCount = random.Next(0, averageDependenciesPerTask * 2);
            for (int j = 0; j < dependencyCount; j++)
            {
                // 只依赖ID较小的任务，避免循环依赖
                int dependencyIndex = random.Next(0, i);
                TaskScheduler.Instance.AddDependency(tasks[i], tasks[dependencyIndex]);
            }
        }
        
        stopwatch.Stop();
        Debug.Log($"创建{taskCount}个任务和依赖关系耗时: {stopwatch.ElapsedMilliseconds}ms");
        
        // 检查循环依赖
        stopwatch.Restart();
        bool hasCircular = TaskScheduler.Instance.HasCircularDependency(out List<string> circularPath);
        stopwatch.Stop();
        Debug.Log($"检查循环依赖耗时: {stopwatch.ElapsedMilliseconds}ms, 是否有循环依赖: {hasCircular}");
        
        // 拓扑排序
        stopwatch.Restart();
        var sortedTasks = TaskScheduler.Instance.GetExecutionOrder();
        stopwatch.Stop();
        Debug.Log($"拓扑排序耗时: {stopwatch.ElapsedMilliseconds}ms");
        
        // 提交任务图并等待完成
        stopwatch.Restart();
        var completionEvent = new ManualResetEvent(false);
        int completedCount = 0;
        
        foreach (var task in tasks)
        {
            task.OnStatusChanged += (t, oldStatus, newStatus) => {
                if (newStatus == TaskStatus.Completed || newStatus == TaskStatus.Failed)
                {
                    if (Interlocked.Increment(ref completedCount) == tasks.Count)
                    {
                        completionEvent.Set();
                    }
                }
            };
        }
        
        TaskScheduler.Instance.SubmitTaskGraph(tasks.ToArray());
        completionEvent.WaitOne();
        
        stopwatch.Stop();
        Debug.Log($"执行{taskCount}个任务耗时: {stopwatch.ElapsedMilliseconds}ms");
        
        // 统计结果
        int completedTasks = tasks.Count(t => t.Status == TaskStatus.Completed);
        int failedTasks = tasks.Count(t => t.Status == TaskStatus.Failed);
        Debug.Log($"任务执行结果: 完成{completedTasks}个, 失败{failedTasks}个");
    }
}
```

本设计方案实现了一个全面的任务依赖关系支持功能，具有以下特点：
