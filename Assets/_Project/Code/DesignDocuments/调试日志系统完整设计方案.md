# 调试日志系统完整设计方案

## 一、需求分析

根据项目架构文档，基础设施层需要包含**日志调试**功能，具体要求：
- **多级别日志**：支持不同严重程度的日志记录
- **内嵌调试窗口**：在游戏内实时显示日志信息
- **高内聚、低耦合**：与其他系统解耦，易于扩展和维护
- **性能优化**：确保日志系统不影响游戏性能
- **支持未来扩展**：为热更新等功能预留接口

## 二、系统设计

### 2.1 架构定位

- **层级**：基础设施层
- **依赖关系**：仅依赖Unity引擎API，不依赖其他业务模块
- **设计模式**：Singleton单例模式 + 观察者模式 + 策略模式

#### 2.3.4 LogManager 核心实现

```csharp
using System;
using System.Collections.Generic;
using UnityEngine;
using Basement.Utils;

namespace Basement.Logging
{
    public class LogManager : Singleton<LogManager>
    {
        private List<ILogOutput> _outputs = new List<ILogOutput>();
        private LogLevel _currentLevel = LogLevel.Debug;
        private bool _isInitialized = false;

        public void Initialize(LogConfig config)
        {
            if (_isInitialized)
                return;

            _currentLevel = config.DefaultLogLevel;
            
            // 添加默认输出
            if (config.EnableConsoleOutput)
                _outputs.Add(new ConsoleOutput());
            
            if (config.EnableFileOutput)
                _outputs.Add(new FileOutput(config.LogFilePath));
            
            if (config.EnableDebugWindow)
            {
                var debugWindow = DebugWindow.Instance;
                debugWindow.MaxLines = config.DebugWindowMaxLines;
                _outputs.Add(debugWindow);
            }
            
            // 设置输出级别
            foreach (var output in _outputs)
            {
                output.SetLogLevel(_currentLevel);
            }
            
            _isInitialized = true;
            Log("日志系统初始化完成", LogLevel.Info, "LoggingSystem");
        }

        public void Log(string message, LogLevel level = LogLevel.Info, string tag = null)
        {
            if (level < _currentLevel)
                return;

            string logMessage = FormatLogMessage(message, level, tag);
            
            foreach (var output in _outputs)
            {
                try
                {
                    output.Log(logMessage, level, tag);
                }
                catch (Exception e)
                {
                    Debug.LogError($"日志输出失败: {e.Message}");
                }
            }
        }

        public void LogDebug(string message, string tag = null)
        { Log(message, LogLevel.Debug, tag); }

        public void LogInfo(string message, string tag = null)
        { Log(message, LogLevel.Info, tag); }

        public void LogWarning(string message, string tag = null)
        { Log(message, LogLevel.Warning, tag); }

        public void LogError(string message, string tag = null)
        { Log(message, LogLevel.Error, tag); }

        public void LogFatal(string message, string tag = null)
        { Log(message, LogLevel.Fatal, tag); }

        private string FormatLogMessage(string message, LogLevel level, string tag)
        {
            string time = DateTime.Now.ToString("HH:mm:ss.fff");
            string levelStr = level.ToString().ToUpper();
            string tagStr = string.IsNullOrEmpty(tag) ? "[GENERAL]" : $"[{tag}]";
            return $"[{time}] {levelStr} {tagStr} {message}";
        }

        public void AddOutput(ILogOutput output)
        {
            if (!_outputs.Contains(output))
            {
                output.SetLogLevel(_currentLevel);
                _outputs.Add(output);
            }
        }

        public void RemoveOutput(ILogOutput output)
        {
            if (_outputs.Contains(output))
            {
                _outputs.Remove(output);
            }
        }

        public void SetLogLevel(LogLevel level)
        {
            _currentLevel = level;
            foreach (var output in _outputs)
            {
                output.SetLogLevel(level);
            }
        }

        protected override void Awake()
        {
            base.Awake();
            Initialize(new LogConfig());
        }
    }
}
```

#### 2.3.5 ConsoleOutput 实现

```csharp
using UnityEngine;

namespace Basement.Logging
{
    public class ConsoleOutput : ILogOutput
    {
        private LogLevel _logLevel = LogLevel.Debug;

        public void Log(string message, LogLevel level, string tag = null)
        {
            if (level < _logLevel)
                return;

            switch (level)
            {
                case LogLevel.Debug:
                case LogLevel.Info:
                    Debug.Log(message);
                    break;
                case LogLevel.Warning:
                    Debug.LogWarning(message);
                    break;
                case LogLevel.Error:
                case LogLevel.Fatal:
                    Debug.LogError(message);
                    break;
            }
        }

        public void SetLogLevel(LogLevel level)
        {
            _logLevel = level;
        }
    }
}
```

#### 2.3.6 FileOutput 实现

```csharp
using System;
using System.IO;
using UnityEngine;

namespace Basement.Logging
{
    public class FileOutput : ILogOutput
    {
        private LogLevel _logLevel = LogLevel.Debug;
        private string _filePath;
        private StreamWriter _writer;
        private object _lock = new object();

        public FileOutput(string filePath = null)
        {
            _filePath = filePath ?? GetDefaultLogPath();
            InitializeFile();
        }

        private string GetDefaultLogPath()
        {
            string logDir = Path.Combine(Application.persistentDataPath, "Logs");
            Directory.CreateDirectory(logDir);
            return Path.Combine(logDir, $"game_{DateTime.Now:yyyyMMdd_HHmmss}.log");
        }

        private void InitializeFile()
        {
            try
            {
                _writer = new StreamWriter(_filePath, true) { AutoFlush = true };
                Log("日志文件初始化完成", LogLevel.Info, "FileOutput");
            }
            catch (Exception e)
            {
                Debug.LogError($"创建日志文件失败: {e.Message}");
            }
        }

        public void Log(string message, LogLevel level, string tag = null)
        {
            if (level < _logLevel || _writer == null)
                return;

            lock (_lock)
            {
                try
                {
                    _writer.WriteLine(message);
                }
                catch (Exception e)
                {
                    Debug.LogError($"写入日志文件失败: {e.Message}");
                }
            }
        }

        public void SetLogLevel(LogLevel level)
        {
            _logLevel = level;
        }
    }
}
```

#### 2.3.7 DebugWindow 内嵌调试窗口

```csharp
using System;
using System.Collections.Generic;
using UnityEngine;
using Basement.Utils;

namespace Basement.Logging
{
    public class DebugWindow : Singleton<DebugWindow>, ILogOutput
    {
        private LogLevel _logLevel = LogLevel.Debug;
        private List<string> _logLines = new List<string>();
        private Vector2 _scrollPosition;
        private bool _isVisible = false;
        private bool _autoScroll = true;
        private LogLevel _filterLevel = LogLevel.Debug;
        private string _searchText = "";
        private GUIStyle _logStyle;
        private GUIStyle _debugStyle;
        private GUIStyle _infoStyle;
        private GUIStyle _warningStyle;
        private GUIStyle _errorStyle;
        private GUIStyle _fatalStyle;
        private int _maxLines = 1000;
        public int MaxLines
        {
            get { return _maxLines; }
            set { _maxLines = Math.Max(100, value); }
        }

        private void OnGUI()
        {
            if (!_isVisible)
                return;

            if (_logStyle == null)
                InitializeStyles();

            // 调试窗口
            GUILayout.Window(0, new Rect(10, 10, Screen.width - 20, Screen.height - 20), DrawDebugWindow, "调试日志");
        }

        private void DrawDebugWindow(int windowId)
        {
            // 工具栏
            GUILayout.BeginHorizontal(EditorStyles.toolbar);
            
            _isVisible = GUILayout.Toggle(_isVisible, "显示", EditorStyles.toolbarButton, GUILayout.Width(50));
            GUILayout.Space(10);
            
            _autoScroll = GUILayout.Toggle(_autoScroll, "自动滚动", EditorStyles.toolbarButton);
            GUILayout.Space(10);
            
            _filterLevel = (LogLevel)GUILayout.Toolbar((int)_filterLevel, 
                Enum.GetNames(typeof(LogLevel)), EditorStyles.toolbarButton);
            
            GUILayout.FlexibleSpace();
            
            if (GUILayout.Button("清空", EditorStyles.toolbarButton))
                _logLines.Clear();
            
            if (GUILayout.Button("关闭", EditorStyles.toolbarButton))
                _isVisible = false;
            
            GUILayout.EndHorizontal();

            // 搜索框
            GUILayout.BeginHorizontal();
            GUILayout.Label("搜索:", GUILayout.Width(50));
            _searchText = GUILayout.TextField(_searchText);
            GUILayout.EndHorizontal();

            // 日志内容
            _scrollPosition = GUILayout.BeginScrollView(_scrollPosition, false, true);
            
            foreach (var line in _logLines)
            {
                if (!string.IsNullOrEmpty(_searchText) && !line.Contains(_searchText))
                    continue;

                if (line.Contains("[DEBUG]"))
                    GUILayout.Label(line, _debugStyle);
                else if (line.Contains("[INFO]"))
                    GUILayout.Label(line, _infoStyle);
                else if (line.Contains("[WARNING]"))
                    GUILayout.Label(line, _warningStyle);
                else if (line.Contains("[ERROR]"))
                    GUILayout.Label(line, _errorStyle);
                else if (line.Contains("[FATAL]"))
                    GUILayout.Label(line, _fatalStyle);
                else
                    GUILayout.Label(line, _logStyle);
            }
            
            GUILayout.EndScrollView();

            // 自动滚动
            if (_autoScroll)
                _scrollPosition.y = float.MaxValue;

            // 允许拖动窗口
            GUI.DragWindow(new Rect(0, 0, 10000, 20));
        }

        private void InitializeStyles()
        {
            _logStyle = new GUIStyle(GUI.skin.label)
            { richText = true, wordWrap = true, fontSize = 12 };
            
            _debugStyle = new GUIStyle(_logStyle)
            { normal = { textColor = Color.gray } };
            
            _infoStyle = new GUIStyle(_logStyle)
            { normal = { textColor = Color.white } };
            
            _warningStyle = new GUIStyle(_logStyle)
            { normal = { textColor = Color.yellow } };
            
            _errorStyle = new GUIStyle(_logStyle)
            { normal = { textColor = Color.red } };
            
            _fatalStyle = new GUIStyle(_logStyle)
            { normal = { textColor = Color.magenta } };
        }

        public void Log(string message, LogLevel level, string tag = null)
        {
            if (level < _logLevel)
                return;

            _logLines.Add(message);
            
            // 限制日志行数
            if (_logLines.Count > _maxLines)
                _logLines.RemoveAt(0);
        }

        public void SetLogLevel(LogLevel level)
        {
            _logLevel = level;
        }

        // 快捷键切换可见性
        private void Update()
        {
            if (Input.GetKeyDown(KeyCode.F12))
                _isVisible = !_isVisible;
        }
    }
}
```

## 四、性能优化

1. **日志级别过滤**：在LogManager层面进行级别过滤，减少不必要的处理
2. **批量写入**：文件输出可以考虑批量写入，减少I/O操作
3. **对象池**：对于频繁创建的日志对象使用对象池
4. **条件编译**：发布版本可以通过条件编译移除Debug级别日志
5. **异步处理**：文件写入可以采用异步方式，避免阻塞主线程

## 五、扩展建议

1. **支持热更新**：可以将日志配置改为可热更新的形式
2. **网络日志**：添加网络输出，将日志发送到服务器
3. **日志分析**：添加日志分析工具，自动识别问题
4. **性能监控**：集成性能监控功能，记录帧率、内存等信息
5. **多语言支持**：添加多语言支持，便于国际化

## 六、总结

本设计方案实现了一个功能完整、性能优良的调试日志系统，符合项目架构文档中的要求。系统采用分层设计，支持多级别日志和内嵌调试窗口，具有良好的可扩展性和可维护性。通过使用单例模式和观察者模式，确保了系统的高内聚、低耦合特性，能够很好地满足游戏开发和调试的需求。
