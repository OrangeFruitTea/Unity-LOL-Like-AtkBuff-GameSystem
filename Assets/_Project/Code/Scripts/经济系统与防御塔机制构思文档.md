# 经济系统与防御塔机制构思文档

## 1. 系统概述

经济系统与防御塔机制是MOBA游戏的核心玩法系统，负责管理游戏中的资源获取、装备购买和防御塔战斗。经济系统确保玩家能够通过击杀小兵、野怪、英雄和推塔获得金币，购买装备提升能力。防御塔机制则为游戏提供战略纵深和攻防玩法。

```csharp
using System;
using System.Collections.Generic;

namespace Gameplay.Economy
{
    /// <summary>
    /// 经济系统
    /// 管理游戏中的金币和经验
    /// </summary>
    public class EconomySystem
    {
        private Dictionary<CharacterBase, PlayerEconomyData> _playerEconomyData;
        
        /// <summary>
        /// 金币变更事件
        /// </summary>
        public event Action<CharacterBase, int, int> OnGoldChanged;
        
        /// <summary>
        /// 经验变更事件
        /// </summary>
        public event Action<CharacterBase, int, int> OnExperienceChanged;
        
        public EconomySystem()
        {
            _playerEconomyData = new Dictionary<CharacterBase, PlayerEconomyData>();
        }
        
        /// <summary>
        /// 初始化玩家经济数据
        /// </summary>
        public void InitializePlayerEconomy(CharacterBase player)
        {
            if (!_playerEconomyData.ContainsKey(player))
            {
                _playerEconomyData[player] = new PlayerEconomyData();
            }
        }
        
        /// <summary>
        /// 获取玩家金币
        /// </summary>
        public int GetPlayerGold(CharacterBase player)
        {
            if (_playerEconomyData.TryGetValue(player, out var data))
            {
                return data.Gold;
            }
            return 0;
        }
        
        /// <summary>
        /// 获取玩家经验
        /// </summary>
        public int GetPlayerExperience(CharacterBase player)
        {
            if (_playerEconomyData.TryGetValue(player, out var data))
            {
                return data.Experience;
            }
            return 0;
        }
        
        /// <summary>
        /// 增加金币
        /// </summary>
        public void AddGold(CharacterBase player, int amount)
        {
            if (_playerEconomyData.TryGetValue(player, out var data))
            {
                int oldGold = data.Gold;
                data.Gold += amount;
                OnGoldChanged?.Invoke(player, oldGold, data.Gold);
            }
        }
        
        /// <summary>
        /// 减少金币
        /// </summary>
        public bool RemoveGold(CharacterBase player, int amount)
        {
            if (_playerEconomyData.TryGetValue(player, out var data))
            {
                if (data.Gold >= amount)
                {
                    int oldGold = data.Gold;
                    data.Gold -= amount;
                    OnGoldChanged?.Invoke(player, oldGold, data.Gold);
                    return true;
                }
            }
            return false;
        }
        
        /// <summary>
        /// 增加经验
        /// </summary>
        public void AddExperience(CharacterBase player, int amount)
        {
            if (_playerEconomyData.TryGetValue(player, out var data))
            {
                int oldExperience = data.Experience;
                data.Experience += amount;
                OnExperienceChanged?.Invoke(player, oldExperience, data.Experience);
                
                // 检查是否升级
                CheckForLevelUp(player, data);
            }
        }
        
        /// <summary>
        /// 检查升级
        /// </summary>
        private void CheckForLevelUp(CharacterBase player, PlayerEconomyData data)
        {
            int currentLevel = player.Level;
            int requiredExperience = GetRequiredExperience(currentLevel);
            
            while (data.Experience >= requiredExperience && currentLevel < player.AttributeComponent.GetMaxLevel())
            {
                data.Experience -= requiredExperience;
                player.LevelUp();
                currentLevel = player.Level;
                requiredExperience = GetRequiredExperience(currentLevel);
            }
        }
        
        /// <summary>
        /// 获取升级所需经验
        /// </summary>
        private int GetRequiredExperience(int level)
        {
            // 经验计算公式
            return 100 * level * level;
        }
        
        /// <summary>
        /// 处理小兵击杀奖励
        /// </summary>
        public void HandleMinionKill(CharacterBase killer, MinionBase minion, bool isLastHit)
        {
            int goldReward = GetMinionGoldReward(minion, isLastHit);
            int expReward = GetMinionExpReward(minion);
            
            AddGold(killer, goldReward);
            AddExperience(killer, expReward);
            
            // 处理范围内的队友经验分享
            ShareExperienceWithNearbyAllies(killer, expReward, 15f);
        }
        
        /// <summary>
        /// 处理英雄击杀奖励
        /// </summary>
        public void HandleHeroKill(CharacterBase killer, CharacterBase victim)
        {
            int goldReward = GetHeroGoldReward(victim);
            int expReward = GetHeroExpReward(victim);
            
            AddGold(killer, goldReward);
            AddExperience(killer, expReward);
            
            // 处理助攻奖励
            List<CharacterBase> assistants = GetAssistants(victim, 10f);
            foreach (var assistant in assistants)
            {
                int assistGold = goldReward / 3;
                int assistExp = expReward / 2;
                AddGold(assistant, assistGold);
                AddExperience(assistant, assistExp);
            }
        }
        
        /// <summary>
        /// 处理防御塔摧毁奖励
        /// </summary>
        public void HandleTowerDestroy(CharacterBase player, TowerBase tower)
        {
            int goldReward = GetTowerGoldReward(tower);
            AddGold(player, goldReward);
            
            // 团队奖励
            ShareGoldWithTeam(player, goldReward / 2);
        }
        
        /// <summary>
        /// 获取小兵金币奖励
        /// </summary>
        private int GetMinionGoldReward(MinionBase minion, bool isLastHit)
        {
            int baseReward = 20;
            // 根据小兵类型和游戏时间调整奖励
            return isLastHit ? baseReward : baseReward / 3;
        }
        
        /// <summary>
        /// 获取小兵经验奖励
        /// </summary>
        private int GetMinionExpReward(MinionBase minion)
        {
            int baseReward = 40;
            // 根据小兵类型调整奖励
            return baseReward;
        }
        
        /// <summary>
        /// 获取英雄金币奖励
        /// </summary>
        private int GetHeroGoldReward(CharacterBase hero)
        {
            int baseReward = 300;
            // 根据英雄等级和击杀 streak 调整奖励
            return baseReward;
        }
        
        /// <summary>
        /// 获取英雄经验奖励
        /// </summary>
        private int GetHeroExpReward(CharacterBase hero)
        {
            int baseReward = 100;
            // 根据英雄等级调整奖励
            return baseReward;
        }
        
        /// <summary>
        /// 获取防御塔金币奖励
        /// </summary>
        private int GetTowerGoldReward(TowerBase tower)
        {
            int baseReward = 150;
            // 根据防御塔类型调整奖励
            return baseReward;
        }
        
        /// <summary>
        /// 与附近队友分享经验
        /// </summary>
        private void ShareExperienceWithNearbyAllies(CharacterBase player, int exp, float range)
        {
            // 实现经验分享逻辑
        }

        /// 获取助攻英雄
```

```csharp
using System;
using System.Collections.Generic;

namespace Gameplay.Economy
{
    /// <summary>
    /// 装备系统
    /// 管理游戏中的装备
    /// </summary>
    public class EquipmentSystem
    {
        private Dictionary<CharacterBase, List<EquipmentBase>> _playerEquipment;
        private Dictionary<string, EquipmentConfig> _equipmentConfigs;
        
        /// <summary>
        /// 装备变更事件
        /// </summary>
        public event Action<CharacterBase, EquipmentBase, EquipmentAction> OnEquipmentChanged;
        
        public EquipmentSystem()
        {
            _playerEquipment = new Dictionary<CharacterBase, List<EquipmentBase>>();
            _equipmentConfigs = new Dictionary<string, EquipmentConfig>();
            LoadEquipmentConfigs();
        }
        
        /// <summary>
        /// 加载装备配置
        /// </summary>
        private void LoadEquipmentConfigs()
        {
            // 从配置文件加载装备数据
        }
        
        /// <summary>
        /// 初始化玩家装备
        /// </summary>
        public void InitializePlayerEquipment(CharacterBase player)
        {
            if (!_playerEquipment.ContainsKey(player))
            {
                _playerEquipment[player] = new List<EquipmentBase>();
            }
        }
        
        /// <summary>
        /// 获取玩家装备
        /// </summary>
        public List<EquipmentBase> GetPlayerEquipment(CharacterBase player)
        {
            if (_playerEquipment.TryGetValue(player, out var equipment))
            {
                return equipment;
            }
            return new List<EquipmentBase>();
        }
        
        /// <summary>
        /// 购买装备
        /// </summary>
        public bool PurchaseEquipment(CharacterBase player, string equipmentId, EconomySystem economySystem)
        {
            if (!_equipmentConfigs.TryGetValue(equipmentId, out var config))
                return false;
            
            // 检查金币是否足够
            if (!economySystem.RemoveGold(player, config.Price))
                return false;
            
            // 创建装备
            EquipmentBase equipment = CreateEquipment(config);
            
            // 检查装备栏是否已满
            if (_playerEquipment[player].Count >= 6)
                return false;
            
            // 添加装备
            _playerEquipment[player].Add(equipment);
            ApplyEquipmentEffects(player, equipment);
            OnEquipmentChanged?.Invoke(player, equipment, EquipmentAction.Purchase);
            
            return true;
        }
        
        /// <summary>
        /// 出售装备
        /// </summary>
        public bool SellEquipment(CharacterBase player, EquipmentBase equipment, EconomySystem economySystem)
        {
            if (!_playerEquipment[player].Contains(equipment))
                return false;
            
            // 移除装备效果
            RemoveEquipmentEffects(player, equipment);
            
            // 退还金币
            int sellPrice = equipment.Config.Price / 2;
            economySystem.AddGold(player, sellPrice);
            
            // 移除装备
            _playerEquipment[player].Remove(equipment);
            OnEquipmentChanged?.Invoke(player, equipment, EquipmentAction.Sell);
            
            return true;
        }
        
        /// <summary>
        /// 合成装备
        /// </summary>
        public bool CombineEquipment(CharacterBase player, List<string> componentIds, string resultId, EconomySystem economySystem)
        {
            // 检查是否拥有所有合成材料
            foreach (var componentId in componentIds)
            {
                if (!HasEquipment(player, componentId))
                    return false;
            }
            
            // 检查合成结果是否存在
            if (!_equipmentConfigs.TryGetValue(resultId, out var resultConfig))
                return false;
            
            // 移除合成材料
            foreach (var componentId in componentIds)
            {
                EquipmentBase component = GetEquipment(player, componentId);
                if (component != null)
                {
                    RemoveEquipmentEffects(player, component);
                    _playerEquipment[player].Remove(component);
                }
            }
            
            // 创建合成结果
            EquipmentBase result = CreateEquipment(resultConfig);
            _playerEquipment[player].Add(result);
            ApplyEquipmentEffects(player, result);
            OnEquipmentChanged?.Invoke(player, result, EquipmentAction.Combine);
            
            return true;
        }
        
        /// <summary>
        /// 创建装备
        /// </summary>
        private EquipmentBase CreateEquipment(EquipmentConfig config)
        {
            return new EquipmentBase(config);
        }
        
        /// <summary>
        /// 应用装备效果
        /// </summary>
        private void ApplyEquipmentEffects(CharacterBase player, EquipmentBase equipment)
        {
            // 应用装备属性加成
            foreach (var attr in equipment.Config.AttributeBonuses)
            {
                player.AttributeComponent.AddModifier(attr.Type, new AttributeModifier
                {
                    Id = equipment.Config.EquipmentId,
                    Value = attr.Value,
                    Type = ModifierType.Flat,
                    Duration = 0
                });
            }
            
            // 应用装备被动效果
            foreach (var passive in equipment.Config.PassiveEffects)
            {
                // 应用被动效果
            }
        }
        
        /// <summary>
        /// 移除装备效果
        /// </summary>
        private void RemoveEquipmentEffects(CharacterBase player, EquipmentBase equipment)
        {
            // 移除装备属性加成
            foreach (var attr in equipment.Config.AttributeBonuses)
            {
                player.AttributeComponent.RemoveModifier(attr.Type, equipment.Config.EquipmentId);
            }
            
            // 移除装备被动效果
            foreach (var passive in equipment.Config.PassiveEffects)
            {
                // 移除被动效果
            }
        }
        
        /// <summary>
        /// 检查玩家是否拥有指定装备
        /// </summary>
        private bool HasEquipment(CharacterBase player, string equipmentId)
        {
            if (_playerEquipment.TryGetValue(player, out var equipmentList))
            {
                return equipmentList.Exists(e => e.Config.EquipmentId == equipmentId);
            }
            return false;
        }
        
        /// <summary>
        /// 获取玩家的指定装备
        /// </summary>
        private EquipmentBase GetEquipment(CharacterBase player, string equipmentId)
        {
            if (_playerEquipment.TryGetValue(player, out var equipmentList))
            {
                return equipmentList.Find(e => e.Config.EquipmentId == equipmentId);
            }
            return null;
        }
    }
    
    /// <summary>
    /// 装备基础类
    /// </summary>
    public class EquipmentBase
    {
        public EquipmentConfig Config { get; protected set; }
        
        public EquipmentBase(EquipmentConfig config)
        {
            Config = config;
        }
    }
    
    /// <summary>
    /// 装备配置
    /// </summary>
    public class EquipmentConfig
    {
        public string EquipmentId { get; set; }
        public string EquipmentName { get; set; }
        public int Price { get; set; }
        public List<AttributeBonus> AttributeBonuses { get; set; }
        public List<PassiveEffect> PassiveEffects { get; set; }
        public List<string> ComponentIds { get; set; }
    }
    
    /// <summary>
    /// 属性加成
    /// </summary>
    public class AttributeBonus
    {
        public AttributeType Type { get; set; }
        public float Value { get; set; }
    }
    
    /// <summary>
    /// 被动效果
    /// </summary>
    public class PassiveEffect
    {
        public string EffectId { get; set; }
        public string EffectName { get; set; }
        public string Description { get; set; }
    }
    
    /// <summary>
    /// 装备操作
    /// </summary>
    public enum EquipmentAction
    {
        Purchase,    // 购买
        Sell,        // 出售
        Combine,     // 合成
        Upgrade      // 升级
    }
}
```

```csharp
using System;
using System.Collections.Generic;
using Basement.Tasks;

namespace Gameplay.Towers
{
    /// <summary>
    /// 防御塔基础类
    /// </summary>
    public class TowerBase : CharacterBase
    {
        /// <summary>
        /// 防御塔类型
        /// </summary>
        public TowerType TowerType { get; protected set; }
        
        /// <summary>
        /// 所属阵营
        /// </summary>
        public TeamSide TeamSide { get; protected set; }
        
        /// <summary>
        /// AI组件
        /// </summary>
        public TowerAIComponent AIComponent { get; protected set; }
        
        /// <summary>
        /// 初始化防御塔
        /// </summary>
        public override void Initialize(string towerId)
        {
            base.Initialize(towerId);
            AIComponent = new TowerAIComponent(this);
        }
        
        /// <summary>
        /// 设置防御塔属性
        /// </summary>
        public virtual void SetTowerProperties(TowerType type, TeamSide side)
        {
            TowerType = type;
            TeamSide = side;
        }
        
        /// <summary>
        /// 更新防御塔
        /// </summary>
        public override void Update()
        {
            base.Update();
            AIComponent?.Update();
        }
        
        /// <summary>
        /// 防御塔被摧毁
        /// </summary>
        public override void OnDeath()
        {
            base.OnDeath();
            // 处理防御塔摧毁逻辑
        }
    }
    
    /// <summary>
    /// 防御塔AI组件
    /// </summary>
    public class TowerAIComponent
    {
        private TowerBase _owner;
        private CharacterBase _target;
        private float _attackRange;
        private float _attackInterval;
        private float _lastAttackTime;
        private float _aggroRange;
        
        public TowerAIComponent(TowerBase owner)
        {
            _owner = owner;
            _attackRange = 700f; // 防御塔攻击范围
            _attackInterval = 1.0f; // 攻击间隔
            _aggroRange = 800f; // 仇恨范围
        }
        
        /// <summary>
        /// 更新AI
        /// </summary>
        public void Update()
        {
            // 检查目标是否有效
            if (_target == null || _target.StateComponent.HasState(CharacterState.Dead))
            {
                _target = FindTarget();
            }
            
            // 攻击目标
            if (_target != null)
            {
                float distance = Vector3.Distance(_owner.transform.position, _target.transform.position);
                if (distance <= _attackRange)
                {
                    if (Time.time - _lastAttackTime >= _attackInterval)
                    {
                        AttackTarget(_target);
                        _lastAttackTime = Time.time;
                    }
                }
                else if (distance > _aggroRange)
                {
                    _target = null;
                }
            }
        }
        
        /// <summary>
        /// 寻找目标
        /// </summary>
        private CharacterBase FindTarget()
        {
            List<CharacterBase> potentialTargets = GetCharactersInRange(_owner.transform.position, _aggroRange);
            
            // 按优先级排序目标
            potentialTargets.Sort((a, b) => GetTargetPriority(a).CompareTo(GetTargetPriority(b)));
            
            return potentialTargets.Count > 0 ? potentialTargets[0] : null;
        }
        
        /// <summary>
        /// 获取目标优先级
        /// </summary>
        private int GetTargetPriority(CharacterBase target)
        {
            // 1. 敌方英雄
            if (target is HeroBase && target.TeamSide != _owner.TeamSide)
                return 1;
            
            // 2. 敌方小兵
            if (target is MinionBase && target.TeamSide != _owner.TeamSide)
                return 2;
            
            // 3. 敌方野怪
            if (target is MonsterBase && target.TeamSide != _owner.TeamSide)
                return 3;
            
            return int.MaxValue;
        }
        
        /// <summary>
        /// 获取范围内的角色
        /// </summary>
        private List<CharacterBase> GetCharactersInRange(Vector3 position, float range)
        {
            // 实现范围检测逻辑
            return new List<CharacterBase>();
        }
        
        /// <summary>
        /// 攻击目标
        /// </summary>
        private void AttackTarget(CharacterBase target)
        {
            // 执行攻击逻辑
            DamageData damageData = new DamageData
            {
                Attacker = _owner,
                Target = target,
                BaseDamage = 100,
                DamageType = DamageType.Physical,
                DamageMultiplier = 1.0f
            };
            
            CombatSystem.ApplyDamage(damageData);
        }
    }
    
    /// <summary>
    /// 防御塔类型
    /// </summary>
    public enum TowerType
    {
        Outer,      // 外塔
        Inner,      // 内塔
        Inhibitor,  // 高地塔
        Nexus       // 基地塔
    }
}
```
