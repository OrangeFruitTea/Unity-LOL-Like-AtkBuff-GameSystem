# 角色与属性系统构思文档

## 1. 系统概述

角色与属性系统是MOBA游戏的核心基础系统，负责管理游戏中所有可交互实体（英雄、小兵、野怪、防御塔等）的属性数据和状态管理。该系统基于EngineAdapterModule封装角色对象管理逻辑，实现角色属性的动态管理与状态同步机制。

## 3. 核心组件

### 3.1 角色基础类

```csharp
using System;
using System.Collections.Generic;

namespace Gameplay.Entity
{
    /// <summary>
    /// 角色基础类
    /// 所有游戏实体的基类
    /// </summary>
    public abstract class CharacterBase : EntityBase
    {
        /// <summary>
        /// 角色属性组件
        /// </summary>
        public CharacterAttributeComponent AttributeComponent { get; protected set; }
        
        /// <summary>
        /// 角色状态组件
        /// </summary>
        public CharacterStateComponent StateComponent { get; protected set; }
        
        /// <summary>
        /// 角色等级
        /// </summary>
        public int Level { get; protected set; }
        
        /// <summary>
        /// 初始化角色
        /// </summary>
        public virtual void Initialize(string characterId)
        {
            base.Initialize(characterId);
            AttributeComponent = new CharacterAttributeComponent(this);
            StateComponent = new CharacterStateComponent(this);
            Level = 1;
        }
        
        /// <summary>
        /// 升级
        /// </summary>
        public virtual void LevelUp()
        {
            Level++;
            AttributeComponent.ApplyLevelGrowth(Level);
            OnLevelUp();
        }
        
        /// <summary>
        /// 升级回调
        /// </summary>
        protected virtual void OnLevelUp() { }
        
        /// <summary>
        /// 更新角色
        /// </summary>
        public override void Update()
        {
            base.Update();
            AttributeComponent?.Update();
            StateComponent?.Update();
        }
    }
}
```

### 3.2 属性组件

```csharp
using System;
using System.Collections.Generic;

namespace Gameplay.Entity
{
    /// <summary>
    /// 角色属性组件
    /// 管理角色的所有属性数据
    /// </summary>
    public class CharacterAttributeComponent
    {
        private CharacterBase _owner;
        private Dictionary<AttributeType, AttributeData> _attributes;
        private Dictionary<AttributeType, List<AttributeModifier>> _modifiers;
        
        /// <summary>
        /// 属性变更事件
        /// </summary>
        public event Action<AttributeType, float, float> OnAttributeChanged;
        
        public CharacterAttributeComponent(CharacterBase owner)
        {
            _owner = owner;
            _attributes = new Dictionary<AttributeType, AttributeData>();
            _modifiers = new Dictionary<AttributeType, List<AttributeModifier>>();
        }
        
        /// <summary>
        /// 初始化属性
        /// </summary>
        public void Initialize(AttributeData[] baseAttributes)
        {
            foreach (var attr in baseAttributes)
            {
                _attributes[attr.Type] = attr;
                _modifiers[attr.Type] = new List<AttributeModifier>();
            }
        }
        
        /// <summary>
        /// 获取属性值
        /// </summary>
        public float GetAttributeValue(AttributeType type)
        {
            if (!_attributes.TryGetValue(type, out var attr))
                return 0f;
            
            float baseValue = attr.BaseValue;
            float totalModifier = 0f;
            
            if (_modifiers.TryGetValue(type, out var modifierList))
            {
                foreach (var modifier in modifierList)
                {
                    if (modifier.IsValid())
                    {
                        totalModifier += modifier.Value;
                    }
                }
            }
            
            return baseValue + totalModifier;
        }
        
        /// <summary>
        /// 添加属性 modifier
        /// </summary>
        public void AddModifier(AttributeType type, AttributeModifier modifier)
        {
            if (!_modifiers.TryGetValue(type, out var list))
            {
                list = new List<AttributeModifier>();
                _modifiers[type] = list;
            }
            
            list.Add(modifier);
            NotifyAttributeChanged(type);
        }
        
        /// <summary>
        /// 移除属性 modifier
        /// </summary>
        public void RemoveModifier(AttributeType type, string modifierId)
        {
            if (_modifiers.TryGetValue(type, out var list))
            {
                list.RemoveAll(m => m.Id == modifierId);
                NotifyAttributeChanged(type);
            }
        }
        
        /// <summary>
        /// 应用等级成长
        /// </summary>
        public void ApplyLevelGrowth(int level)
        {
            foreach (var attr in _attributes.Values)
            {
                attr.BaseValue = attr.InitialValue + (attr.GrowthValue * (level - 1));
                NotifyAttributeChanged(attr.Type);
            }
        }
        
        /// <summary>
        /// 通知属性变更
        /// </summary>
        private void NotifyAttributeChanged(AttributeType type)
        {
            float oldValue = _attributes[type].BaseValue;
            float newValue = GetAttributeValue(type);
            OnAttributeChanged?.Invoke(type, oldValue, newValue);
        }
        
        /// <summary>
        /// 更新属性组件
        /// </summary>
        public void Update()
        {
            // 清理过期的modifier
            foreach (var kvp in _modifiers)
            {
                kvp.Value.RemoveAll(m => !m.IsValid());
            }
        }
    }
    
    /// <summary>
    /// 属性类型
    /// </summary>
    public enum AttributeType
    {
        Health,           // 生命值
        Mana,             // 魔法值
        AttackDamage,     // 攻击力
        AbilityPower,     // 法术强度
        AttackSpeed,      // 攻击速度
        MoveSpeed,        // 移动速度
        Armor,            // 物理防御
        MagicResist,      // 魔法抗性
        HealthRegen,      // 生命回复
        ManaRegen,        // 魔法回复
        CriticalChance,   // 暴击率
        CriticalDamage,   // 暴击伤害
        Lifesteal,        // 生命偷取
        SpellVamp,        // 法术吸血
        CooldownReduction // 冷却缩减
    }
    
    /// <summary>
    /// 属性数据
    /// </summary>
    public class AttributeData
    {
        public AttributeType Type { get; set; }
        public float InitialValue { get; set; }
        public float BaseValue { get; set; }
        public float GrowthValue { get; set; }
    }
    
    /// <summary>
    /// 属性修饰符
    /// </summary>
    public class AttributeModifier
    {
        public string Id { get; set; }
        public float Value { get; set; }
        public ModifierType Type { get; set; }
        public float Duration { get; set; }
        public float ElapsedTime { get; set; }
        
        /// <summary>
        /// 检查modifier是否有效
        /// </summary>
        public bool IsValid()
        {
            return Duration <= 0 || ElapsedTime < Duration;
        }
        
        /// <summary>
        /// 更新modifier
        /// </summary>
        public void Update(float deltaTime)
        {
            if (Duration > 0)
            {
                ElapsedTime += deltaTime;
            }
        }
    }
    
    /// <summary>
    /// 修饰符类型
    /// </summary>
    public enum ModifierType
    {
        Flat,       // 固定值
        Percentage  // 百分比
    }
}
```

### 3.3 状态组件

```csharp
using System;
using System.Collections.Generic;

namespace Gameplay.Entity
{
    /// <summary>
    /// 角色状态组件
    /// 管理角色的状态
    /// </summary>
    public class CharacterStateComponent
    {
        private CharacterBase _owner;
        private Dictionary<CharacterState, bool> _states;
        
        /// <summary>
        /// 状态变更事件
        /// </summary>
        public event Action<CharacterState, bool> OnStateChanged;
        
        public CharacterStateComponent(CharacterBase owner)
        {
            _owner = owner;
            _states = new Dictionary<CharacterState, bool>();
            
            // 初始化所有状态为false
            foreach (CharacterState state in Enum.GetValues(typeof(CharacterState)))
            {
                _states[state] = false;
            }
        }
        
        /// <summary>
        /// 设置状态
        /// </summary>
        public void SetState(CharacterState state, bool value)
        {
            if (_states.TryGetValue(state, out var currentValue))
            {
                if (currentValue != value)
                {
                    _states[state] = value;
                    OnStateChanged?.Invoke(state, value);
                }
            }
        }
        
        /// <summary>
        /// 检查状态
        /// </summary>
        public bool HasState(CharacterState state)
        {
            return _states.TryGetValue(state, out var value) && value;
        }
        
        /// <summary>
        /// 清除所有状态
        /// </summary>
        public void ClearAllStates()
        {
            foreach (CharacterState state in Enum.GetValues(typeof(CharacterState)))
            {
                SetState(state, false);
            }
        }
        
        /// <summary>
        /// 更新状态组件
        /// </summary>
        public void Update()
        {
            // 状态更新逻辑
        }
    }
    
    /// <summary>
    /// 角色状态
    /// </summary>
    public enum CharacterState
    {
        Normal,         // 正常
        Stunned,        // 眩晕
        Silenced,       // 沉默
        Rooted,         // 定身
        Snared,         // 减速
        Invulnerable,   // 无敌
        Invisible,      // 隐身
        Dead            // 死亡
    }
}
```

## 5. 技术实现要求

1. **属性管理**：实现角色属性的动态计算，支持临时加成、永久加成等多种属性影响方式
2. **状态管理**：设计角色状态系统，支持多种状态的组合和冲突处理
3. **事件系统**：实现属性变更和状态变更事件，支持外部系统的回调处理
4. **网络同步**：确保客户端与服务端属性数据的一致性
5. **性能优化**：使用对象池和缓存机制，减少频繁创建销毁带来的性能开销

## 10. 软件工程设计与架构分析

### 10.1 设计模式应用分析

#### 10.1.1 创建型设计模式

**工厂模式**
- **应用场景**：角色对象的创建和管理
- **实施要点**：
  - 创建CharacterFactory工厂类，统一管理角色实例化
  - 支持通过角色ID动态创建不同类型的角色实例
  - 封装角色初始化逻辑，包括属性和状态组件的创建
- **预期效益**：降低客户端与具体角色类的耦合度，便于角色类型的扩展

**抽象工厂模式**
- **应用场景**：不同阵营角色族的创建
- **实施要点**：
  - 定义ICharacterFactory接口，包含CreateHero、CreateMinion等方法
  - 实现BlueTeamFactory和RedTeamFactory具体工厂
  - 每个工厂创建对应阵营风格的角色实例
- **预期效益**：支持多阵营角色族的统一管理，便于添加新阵营

**建造者模式**
- **应用场景**：复杂角色对象的构建
- **实施要点**：
  - 创建CharacterBuilder类，支持分步构建角色
  - 提供SetAttributes、SetStates、SetEquipment等方法
  - 支持角色配置的灵活组合
- **预期效益**：简化复杂对象的创建过程，提高代码可读性

#### 10.1.2 结构型设计模式

**适配器模式**
- **应用场景**：EngineAdapterModule模块核心逻辑封装
- **实施要点**：
  - 创建IEngineAdapter接口，定义引擎操作的标准方法
  - 实现UnityEngineAdapter类，适配Unity引擎的具体实现
  - 提供统一的角色生命周期管理接口
- **预期效益**：隔离引擎依赖，便于引擎替换和跨平台移植

**外观模式**
- **应用场景**：角色管理系统的简化接口
- **实施要点**：
  - 创建CharacterManagerFacade类，提供简化的角色管理接口
  - 封装内部复杂的属性、状态、事件系统操作
  - 提供CreateCharacter、DestroyCharacter、UpdateCharacter等高层方法
- **预期效益**：简化客户端使用，降低系统复杂度

**装饰器模式**
- **应用场景**：角色属性修饰符的动态添加
- **实施要点**：
  - 创建AttributeDecorator抽象装饰器类
  - 实现FlatModifierDecorator和PercentageModifierDecorator具体装饰器
  - 支持修饰符的动态组合和嵌套
- **预期效益**：灵活支持多种属性修饰方式，便于扩展新的修饰类型

**组合模式**
- **应用场景**：角色组件的树形结构管理
- **实施要点**：
  - 创建ICharacterComponent接口
  - 实现CharacterComponentComposite类，支持组件的嵌套组合
  - 支持属性组件、状态组件、装备组件的统一管理
- **预期效益**：统一组件管理接口，支持复杂的组件组合关系

#### 10.1.3 行为型设计模式

**观察者模式**
- **应用场景**：属性变更和状态变更事件系统
- **实施要点**：
  - 创建ICharacterObserver接口，定义OnAttributeChanged、OnStateChanged方法
  - CharacterAttributeComponent和CharacterStateComponent作为Subject
  - 支持多个观察者订阅同一事件
  - 提供AddObserver、RemoveObserver、NotifyObservers方法
- **预期效益**：实现松耦合的事件通知机制，支持多系统协同

**策略模式**
- **应用场景**：属性动态计算引擎
- **实施要点**：
  - 创建IAttributeCalculationStrategy接口，定义Calculate方法
  - 实现FlatCalculationStrategy、PercentageCalculationStrategy等策略
  - 支持运行时切换不同的计算策略
  - 支持自定义属性计算逻辑
- **预期效益**：灵活支持多种属性计算方式，便于扩展新的计算规则

**状态模式**
- **应用场景**：角色状态管理
- **实施要点**：
  - 创建ICharacterState接口，定义Enter、Exit、Update方法
  - 实现NormalState、StunnedState、DeadState等具体状态类
  - CharacterStateComponent维护当前状态，支持状态转换
  - 支持状态转换的前置条件和后置处理
- **预期效益**：简化复杂状态逻辑，提高状态管理的可维护性

**命令模式**
- **应用场景**：角色属性和状态的撤销/重做操作
- **实施要点**：
  - 创建ICharacterCommand接口，定义Execute、Undo方法
  - 实现ChangeAttributeCommand、SetStateCommand等具体命令
  - 创建CommandInvoker类，管理命令的执行和撤销
  - 支持命令队列和批量操作
- **预期效益**：支持操作历史记录和撤销功能，便于调试和回放

**责任链模式**
- **应用场景**：属性修饰符的优先级处理
- **实施要点**：
  - 创建AttributeModifierHandler抽象处理器
  - 实现HighPriorityHandler、MediumPriorityHandler、LowPriorityHandler
  - 支持修饰符处理链的动态构建
  - 每个处理器处理特定优先级的修饰符
- **预期效益**：灵活控制修饰符处理顺序，支持复杂的优先级规则

**访问者模式**
- **应用场景**：角色属性和状态的数据导出和分析
- **实施要点**：
  - 创建ICharacterVisitor接口，定义VisitAttribute、VisitState方法
  - 实现ExportVisitor、AnalysisVisitor等具体访问者
  - 支持对角色数据进行不同类型的处理
- **预期效益**：分离数据结构和操作逻辑，便于添加新的数据处理方式

#### 10.1.4 单例模式

**ConfigurationModule配置模块**
- **应用场景**：全局配置管理
- **实施要点**：
  - 使用单例模式确保ConfigurationModule全局唯一
  - 提供线程安全的配置访问接口
  - 支持配置的延迟加载和缓存
  - 提供配置热更新机制
- **预期效益**：确保配置数据的一致性，减少内存占用

**适用性评估**：
- ✅ 适用于：全局唯一的配置管理器、资源加载器、日志管理器
- ❌ 不适用于：角色实例、属性组件等需要多实例的对象

### 10.2 架构设计考量

#### 10.2.1 模块划分原则与边界定义

**EngineAdapterModule职责**：
- 封装底层引擎操作，提供统一的引擎适配接口
- 管理角色对象的生命周期（创建、更新、销毁）
- 提供引擎相关的工具方法（时间管理、输入处理等）
- **边界定义**：不包含业务逻辑，只负责引擎层面的操作

**ConfigurationModule职责**：
- 管理游戏配置数据的加载、解析和缓存
- 提供配置数据的查询接口
- 支持配置的热更新和版本管理
- **边界定义**：不涉及配置数据的使用逻辑，只负责配置管理

**职责分离原则**：
- **单一职责原则**：每个模块只负责一个明确的职责
- **接口隔离原则**：模块间通过最小化接口进行交互
- **依赖倒置原则**：高层模块不依赖低层模块，都依赖抽象

```csharp
public class AttributeCalculationEngine
{
    public float CalculateFinalValue(AttributeType type, float baseValue)
    {
        // 1. 获取所有修饰符
        var modifiers = GetModifiers(type);
        
        // 2. 按优先级分组
        var groupedModifiers = GroupByPriority(modifiers);
        
        // 3. 按顺序应用修饰符
        float finalValue = baseValue;
        foreach (var group in groupedModifiers)
        {
            finalValue = ApplyModifierGroup(finalValue, group);
        }
        
        // 4. 应用属性限制
        finalValue = ApplyAttributeLimits(type, finalValue);
        
        return finalValue;
    }
    
    private List<ModifierGroup> GroupByPriority(List<AttributeModifier> modifiers)
    {
        // 使用桶排序按优先级分组
        var buckets = new List<ModifierGroup>[10];
        foreach (var modifier in modifiers)
        {
            int priority = modifier.Priority;
            if (buckets[priority] == null)
                buckets[priority] = new List<ModifierGroup>();
            buckets[priority].Add(new ModifierGroup(modifier));
        }
        return buckets.Where(b => b != null).SelectMany(b => b).ToList();
    }
}
```

```csharp
public class AttributeCalculationCache
{
    private Dictionary<string, CacheEntry> _cache;
    private int _maxCacheSize;
    private LinkedList<string> _lruList;
    
    public AttributeCalculationCache(int maxSize = 1000)
    {
        _maxCacheSize = maxSize;
        _cache = new Dictionary<string, CacheEntry>();
        _lruList = new LinkedList<string>();
    }
    
    public bool TryGetCache(string key, out float value)
    {
        if (_cache.TryGetValue(key, out var entry))
        {
            // 更新LRU
            _lruList.Remove(key);
            _lruList.AddFirst(key);
            
            // 检查缓存是否有效
            if (entry.IsValid())
            {
                value = entry.Value;
                return true;
            }
            else
            {
                // 缓存失效，移除
                _cache.Remove(key);
                _lruList.Remove(key);
            }
        }
        
        value = 0;
        return false;
    }
    
    public void SetCache(string key, float value, float ttl = 1.0f)
    {
        // 检查缓存大小
        if (_cache.Count >= _maxCacheSize)
        {
            // 移除最久未使用的缓存
            string lruKey = _lruList.Last.Value;
            _cache.Remove(lruKey);
            _lruList.RemoveLast();
        }
        
        // 添加新缓存
        _cache[key] = new CacheEntry(value, ttl);
        _lruList.AddFirst(key);
    }
    
    public void InvalidateCache(string key)
    {
        if (_cache.ContainsKey(key))
        {
            _cache.Remove(key);
            _lruList.Remove(key);
        }
    }
    
    public void ClearCache()
    {
        _cache.Clear();
        _lruList.Clear();
    }
}
```

**缓存策略**：
- **LRU缓存**：缓存最近使用的属性计算结果
- **TTL机制**：设置缓存有效期，自动清理过期缓存
- **失效策略**：属性变更时主动失效相关缓存
- **批量失效**：支持批量失效相关缓存

#### 10.5.3 高频状态更新的批处理策略

**批处理器实现**：
```csharp
public class StateUpdateBatchProcessor
{
    private List<StateUpdate> _pendingUpdates;
    private List<StateUpdate> _processingUpdates;
    private float _batchInterval;
    private float _lastProcessTime;
    
    public StateUpdateBatchProcessor(float batchInterval = 0.1f)
    {
        _batchInterval = batchInterval;
        _pendingUpdates = new List<StateUpdate>();
        _processingUpdates = new List<StateUpdate>();
    }
    
    public void QueueUpdate(StateUpdate update)
    {
        _pendingUpdates.Add(update);
    }
    
    public void Update(float deltaTime)
    {
        if (Time.time - _lastProcessTime >= _batchInterval)
        {
            ProcessBatch();
            _lastProcessTime = Time.time;
        }
    }
    
    private void ProcessBatch()
    {
        if (_pendingUpdates.Count == 0)
            return;
        
        // 交换列表
        var temp = _processingUpdates;
        _processingUpdates = _pendingUpdates;
        _pendingUpdates = temp;
        _pendingUpdates.Clear();
        
        // 批量处理更新
        ProcessUpdates(_processingUpdates);
        
        // 清空处理列表
        _processingUpdates.Clear();
    }
    
    private void ProcessUpdates(List<StateUpdate> updates)
    {
        // 按角色分组
        var groupedUpdates = updates.GroupBy(u => u.CharacterId);
        
        foreach (var group in groupedUpdates)
        {
            // 合并同一角色的更新
            StateUpdate mergedUpdate = MergeUpdates(group);
            
            // 应用更新
            ApplyUpdate(mergedUpdate);
        }
    }
    
    private StateUpdate MergeUpdates(IGrouping<string, StateUpdate> updates)
    {
        // 合并同一角色的多个更新
        StateUpdate merged = new StateUpdate();
        merged.CharacterId = updates.Key;
        
        foreach (var update in updates)
        {
            // 合并属性更新
            foreach (var attr in update.AttributeUpdates)
            {
                merged.AttributeUpdates[attr.Key] = attr.Value;
            }
            
            // 合并状态更新
            foreach (var state in update.StateUpdates)
            {
                merged.StateUpdates[state.Key] = state.Value;
            }
        }
        
        return merged;
    }
}
```

**批处理优化策略**：
- **时间窗口**：在固定时间窗口内收集更新
- **合并更新**：合并同一角色的多个更新
- **优先级处理**：优先处理高优先级的更新
- **异步处理**：使用异步处理提高性能
